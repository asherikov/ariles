49 class ArilesBaseClass
50  // must inherit from ariles::ConfigurableBase
51  : public ariles::ConfigurableBase
52 {
53  // Unique entry name, to be safe use only alphanumeric characters and underscores
54  #define ARILES_SECTION_ID "ArilesBaseClass"
55 
56  // Declare entries, in this case two numbers
57  #define ARILES_ENTRIES \
58  ARILES_TYPED_ENTRY(real_member, double) \
59  ARILES_TYPED_ENTRY_(integer_member, int)
60  // underscore ^ indicates that the name of the entry must be
61  // 'integer_member_' instead of 'integer_member', this is useful if your
62  // naming convention requires trailining underscores for member variables.
63 
64  // Initialize ariles
65  #include ARILES_INITIALIZE
66 
67  public:
68  virtual ~ArilesBaseClass(){}; // added to suppress compiler warnings
69 
70  // setDefaults() is a method which is called every time you deserialize
71  // a class. You can implement it manually as here, or request its
72  // automatic generation using ARILES_AUTO_DEFAULTS as demonstrated for
73  // other classes below.
74  virtual void setDefaults()
75  {
76  real_member = 0.0;
77  integer_member_ = 12;
78  }
79 };
