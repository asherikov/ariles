38 //! Wrapper of \c std::basic_istream into RapidJSON's Stream concept.
39 /*!
40  The classes can be wrapped including but not limited to:
41 
42  - \c std::istringstream
43  - \c std::stringstream
44  - \c std::wistringstream
45  - \c std::wstringstream
46  - \c std::ifstream
47  - \c std::fstream
48  - \c std::wifstream
49  - \c std::wfstream
50 
51  \tparam StreamType Class derived from \c std::basic_istream.
52 */
53 
54 template <typename StreamType>
55 class BasicIStreamWrapper {
56 public:
57  typedef typename StreamType::char_type Ch;
58  BasicIStreamWrapper(StreamType& stream) : stream_(stream), count_(), peekBuffer_() {}
59 
60  Ch Peek() const {
61  typename StreamType::int_type c = stream_.peek();
62  return (c != StreamType::traits_type::eof()) ? static_cast<Ch>(c) : static_cast<Ch>('\0');
63  }
64 
65  Ch Take() {
66  typename StreamType::int_type c = stream_.get();
67  if (c != StreamType::traits_type::eof()) {
68  count_++;
69  return static_cast<Ch>(c);
70  }
71  else
72  return '\0';
73  }
74 
75  // tellg() may return -1 when failed. So we count by ourself.
76  size_t Tell() const { return count_; }
77 
78  Ch* PutBegin() { RAPIDJSON_ASSERT(false); return 0; }
79  void Put(Ch) { RAPIDJSON_ASSERT(false); }
80  void Flush() { RAPIDJSON_ASSERT(false); }
81  size_t PutEnd(Ch*) { RAPIDJSON_ASSERT(false); return 0; }
82 
83  // For encoding detection only.
84  const Ch* Peek4() const {
85  RAPIDJSON_ASSERT(sizeof(Ch) == 1); // Only usable for byte stream.
86  int i;
87  bool hasError = false;
88  for (i = 0; i < 4; ++i) {
89  typename StreamType::int_type c = stream_.get();
90  if (c == StreamType::traits_type::eof()) {
91  hasError = true;
92  stream_.clear();
93  break;
94  }
95  peekBuffer_[i] = static_cast<Ch>(c);
96  }
97  for (--i; i >= 0; --i)
98  stream_.putback(peekBuffer_[i]);
99  return !hasError ? peekBuffer_ : 0;
100  }
101 
102 private:
103  BasicIStreamWrapper(const BasicIStreamWrapper&);
104  BasicIStreamWrapper& operator=(const BasicIStreamWrapper&);
105 
106  StreamType& stream_;
107  size_t count_; //!< Number of characters read. Note:
108  mutable Ch peekBuffer_[4];
109 };
