109  template <typename t_Scalar>
110  static bool compareFloats(const t_Scalar left, const t_Scalar right, const Parameters &param)
111  {
112  if (isNaN(left))
113  {
114  if (isNaN(right))
115  {
116  return (param.nan_equal_);
117  }
118  else
119  {
120  return (false);
121  }
122  }
123 
124  if (isInfinity(left))
125  {
126  if (isInfinity(right))
127  {
128  if (((left > 0) && (right > 0)) || ((left < 0) && (right < 0)))
129  {
130  return (param.inf_equal_);
131  }
132  }
133  return (false);
134  }
135  return (std::abs(left - right)
136  <= ((std::abs(left) < std::abs(right) ? std::abs(right) : std::abs(left))
137  * param.double_tolerance_));
138  }
