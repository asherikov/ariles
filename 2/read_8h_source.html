55  void checkSize(
56  const SizeLimitEnforcementType limit_type,
57  const std::size_t size = 0,
58  const std::size_t min = 0,
59  const std::size_t max = 0) const
60  {
61  switch (limit_type)
62  {
63  case SIZE_LIMIT_NONE:
64  return;
65  case SIZE_LIMIT_EQUAL:
66  ARILES2_ASSERT(size == min, "Actual number of entries is not the same as expected.");
67  return;
68  case SIZE_LIMIT_RANGE:
69  ARILES2_ASSERT(min <= size, "Actual number of entries is lower than expected.");
70  ARILES2_ASSERT(max >= size, "Actual number of entries is larger than expected.");
71  return;
72  case SIZE_LIMIT_MIN:
73  ARILES2_ASSERT(min <= size, "Actual number of entries is lower than expected.");
74  return;
75  default:
76  ARILES2_THROW("Internal logic error.");
77  return;
78  }
