63  void checkSize(
64  const SizeLimitEnforcementType limit_type,
65  const std::size_t size = 0,
66  const std::size_t min = 0,
67  const std::size_t max = 0) const
68  {
69  switch (limit_type)
70  {
71  case SIZE_LIMIT_NONE:
72  return;
73  case SIZE_LIMIT_EQUAL:
74  ARILES2_ASSERT(size == min, "Actual number of entries is not the same as expected.");
75  return;
76  case SIZE_LIMIT_RANGE:
77  ARILES2_ASSERT(min <= size, "Actual number of entries is lower than expected.");
78  ARILES2_ASSERT(max >= size, "Actual number of entries is larger than expected.");
79  return;
80  case SIZE_LIMIT_MIN:
81  ARILES2_ASSERT(min <= size, "Actual number of entries is lower than expected.");
82  return;
83  default:
84  ARILES2_THROW("Internal logic error.");
85  return;
86  }
